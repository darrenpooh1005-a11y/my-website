                            setTasks(all.filter(t => t.cid === conn.id).sort((a,b)=>b.createdAt - a.createdAt));

                            先略過，隨便看看

                        </button>

                            前往控制台檢查

                        </a>

                    </div>

                    <button onClick={() => window.location.reload()} className="text-blue-500 font-black border-b-2 border-blue-200 pb-1">完成後重新載入</button>

                </div>

            );


            if (view === 'loading') return <div className="h-screen flex items-center justify-center font-black text-slate-300 animate-pulse text-2xl italic tracking-tighter">小島載入中...</div>;


            if (view === 'intro') return (

                        {profile?.role === '島主' ? (

                            <div className="space-y-6">

                                <div className="grid grid-cols-2 gap-4">

                                            if (!connections[0]) return;

                                            addDoc(getPublicPath('tasks'), { 

                                                cid: connections[0].id, 

                                                detail: cat === '派採買' ? '去幫我買點東西！' : '我出門囉，回報位置', 

                                                note: msg, sender: profile.name, status: 'pending', createdAt: Date.now() 

                                            });

                                            setCat(null); setMsg('');

                                </div>

                                {tasks.filter(t=>t.status==='pending').map(t => (

                                    <div key={t.id} className="bg-white p-8 rounded-[3rem] shadow-xl text-center border-t-[12px] border-slate-50 animate-in slide-in-from-bottom-4">

                                        <h2 className="text-5xl font-black text-slate-800 mb-10 italic tracking-tighter leading-none">{t.detail}</h2>

                                        <button onClick={() => updateDoc(getPublicDoc('tasks', t.id), {status: 'arrived'})} className="w-full py-8 bg-emerald-400 text-white rounded-3xl font-black text-2xl shadow-lg active:scale-95 transition-all">我完成了！</button>

                                    </div>

                                )) || <div className="py-20 opacity-10 font-black italic">目前島上很安靜</div>}

                        )}

                    </main>

                </div>

            );
